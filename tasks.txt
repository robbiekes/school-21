Task1.
	Написать функцию int ft_max(unsgined int num, ...)
	которая возвращает максимум из переданных чисел типа int.
	Первый аргумент -- количество чисел, далее идут числа тип int,
	среди которых нужно найти максимум.
	Разрешено: va_arg, va_start, va_end.

Task4.
	Написать функцию void ft_persents(const char *str);
	Которая печатает строку str, но если в строке встретился
	шаблон %d, то печатает также "NUMBER",
	шаблон %u, то печатает также "UNSIGNED",
	шаблон %x, то печатает также "hex",
	шаблон %X, то печатает также "HEX",
	шаблон %p, то печатает также "PTR",
	шаблон %%, то печатает также "%".
	Разрешено, write
Task4.
	Написать функцию int ft_mini_printf(const char *format, ...);
	-- урезанную версию printf, которая поддерживает флаги
	%d
	%i
	%u
	%s
	%c
	%%
	Разрешено, write, va_arg, va_start, va_end.
Task5.
	Написать функцию ft_print_hex(unsigned int num, char is_low);
	которая печатает число num в шестнадцатеричной системе счисления.
	Если аргумент is_low == 1, то символы ABCDEF печатаются в нижнем регистре (abcdef)
	иначе в верхнем регистре.
	Разрешено, write
